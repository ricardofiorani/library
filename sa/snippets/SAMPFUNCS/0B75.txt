{$CLEO}
{$USE SAMPFUNCS}
NOP

repeat
    wait 100
until samp_is_available

int testPfx = allocate_memory {size} 256
int testBdy = allocate_memory {size} 256

int pLast = testPfx + 124
int bLast = testBdy + 124

while true
    int pfxColor, bdyColor
    
    wait {time} 3000
    if samp_get_chat_line {index} 99 {bodyBuffer} testBdy {prefixBuffer} testPfx {bodyColor} bdyColor {prefixColor} pfxColor // get most recent message
    then
        write_memory {address} pLast {size} 1 {value} 0 {vp} 0
        write_memory {address} bLast {size} 1 {value} 0 {vp} 0
        print_formatted_now {format} "Body Color: %p , Message: %s~n~Prefix Color: %p , Message: %s" {time} 100 {args} bdyColor testBdy pfxColor testPfx
    else print_string_now {format} "Test 1 Latest Chat Line is empty" {time} 100
    end
    
    wait {time} 3000
    if samp_get_chat_line {index} 99 {bodyBuffer} testBdy {prefixBuffer} 0 {bodyColor} bdyColor {prefixColor} -1 // get body only
    then
        write_memory {address} bLast {size} 1 {value} 0 {vp} 0
        print_formatted_now {format} "Body Color: %p , Message: %s" {time} 100 {args} bdyColor testBdy
    else print_string_now {format} "Test 2 Latest Chat Line is empty" {time} 100
    end
    
    wait {time} 3000
    if samp_get_chat_line {index} 99 {bodyBuffer} 0 {prefixBuffer} testPfx {bodyColor} -1 {prefixColor} pfxColor // get prefix only
    then
        write_memory {address} pLast {size} 1 {value} 0 {vp} 0
        write_memory {address} bLast {size} 1 {value} 0 {vp} 0
        print_formatted_now {format} "Prefix Color: %p , Message: %s" {time} 100 {args} pfxColor testPfx
    else print_string_now {format} "Test 3 Latest Chat Line is empty" {time} 100
    end
    
    wait {time} 3000
    if samp_get_chat_line {index} 99 {bodyBuffer} testBdy {prefixBuffer} testPfx {bodyColor} -1 {prefixColor} -1 // get strings only
    then
        write_memory {address} pLast {size} 1 {value} 0 {vp} 0
        write_memory {address} bLast {size} 1 {value} 0 {vp} 0
        print_formatted_now {format} "Body Message: %s~n~Prefix Message: %s" {time} 100 {args} testBdy testPfx
    else print_string_now {format} "Test 4 Latest Chat Line is empty" {time} 100
    end
    
    wait {time} 3000
    if samp_get_chat_line {index} 99 {bodyBuffer} 0 {prefixBuffer} 0 {bodyColor} bdyColor {prefixColor} pfxColor // get colors only
    then
        write_memory {address} pLast {size} 1 {value} 0 {vp} 0
        write_memory {address} bLast {size} 1 {value} 0 {vp} 0
        print_formatted_now {format} "Body Color: %p~n~Prefix Color: %p" {time} 100 {args} pfxColor bdyColor
    else print_string_now {format} "Test 5 Latest Chat Line is empty" {time} 100
    end
end
