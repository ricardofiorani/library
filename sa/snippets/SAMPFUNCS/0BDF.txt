// Script Example
// The main thread will execute after the user types "/resumethread"

{$USE CLEO}
{$USE SAMPFUNCS}
NOP

// wait for SAMP to fully initialize
REPEAT
    wait 100
UNTIL SAMP_IS_AVAILABLE

SAMP_REGISTER_CLIENTSIDE_COMMAND {chatcommand} "resumethread" {callbacklabel} @ChatCallback_ResumeThread

SF_PAUSE_THREAD {streamid} 0 // Pauses the main thread

// This part of the main thread will not be reached until SF_RESUME_THREAD is called
while true
    wait 2000
    PRINT_STRING_NOW "Main Loop Executed!" {time} 1000
end


:ChatCallback_ResumeThread
    SF_RESUME_THREAD {threadId} 0 // Resumes this script's main thread
SAMP_CMD_RET // end of callback
