while true
    wait 0

    if
        is_key_pressed {keyCode} KeyCode.F8
    then
        print_help_formatted {text} "~r~Key pressed!"

        
        // wait for key release
        while is_key_pressed {keyCode} KeyCode.F8
            wait 0
        end
        
        print_help_formatted {text} "~g~Key released!"
    end
end
